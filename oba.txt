(function() {
    'use strict';

    // Videoları her 2 saniyede bir kontrol et ve hızlandır
    setInterval(function() {
        let videos = document.querySelectorAll('video');
        videos.forEach(function(video) {
            if (video.readyState >= 2) { // Video yüklendiyse
                video.playbackRate = 10; // Hızı 10 katına çıkar
                video.play(); // Videoyu oynat
            }
        });
    }, 2000); // Her 2 saniyede bir videoları kontrol et
})();


















(function() {
    'use strict';
    window.onblur = () => {};
    var video = document.getElementById("video_html5_api");
    video.muted = true;

    setTimeout(function() {
        video.play();
    }, 5000);

    video.onended = function() {
        console.log("Video bitti, bir sonrakine geçiliyor");
        document.getElementsByClassName("progress-icon")[0].parentElement.click();
        document.getElementById("video_html5_api").play();
    };

    setTimeout(function() {
        location.reload();
    }, 25*60*1000);
})();
