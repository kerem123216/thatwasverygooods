// ==UserScript==
// @name         Video Hızlandırıcı (15 Saniye Sonra)
// @namespace    http://tampermonkey.net/
// @version      1.9
// @description  Videoları 10 kat hızlandır ve tespit edilmeyi önle
// @author       SeninAdın
// @match        *://*/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const speedIncrease = 10; // Hız 10 katına çıkar

    // Sayfa yüklendikten 15 saniye sonra hızlandırmayı başlat
    setTimeout(function() {
        // Videoları her 3 saniyede bir kontrol et
        setInterval(function() {
            let videos = document.querySelectorAll('video');
            videos.forEach(function(video) {
                if (video.readyState >= 2) { // Video yüklendiyse
                    video.playbackRate = speedIncrease; // Hızı artır

                    // Hızlandırmayı gizlemek için görsel ve zamanlama ayarları
                    video.onplay = function() {
                        setTimeout(() => {
                            video.currentTime += 0.1; // Videoyu her 0.1 saniyede bir ilerlet
                        }, 100); // 100 ms aralıklarla ilerlet
                    };

                    // Video bitince işaretle
                    video.onended = function() {
                        console.log("Video tamamlandı."); // Konsolda video tamamlandığını yazdır
                    };

                    // Videoyu oynat
                    video.play().catch(function(error) {
                        console.error("Video oynatmada hata:", error); // Hata loglama
                    });
                }
            });
        }, 3000); // Her 3 saniyede bir videoları kontrol et
    }, 15000); // 15 saniye bekle
})();








(function() {
    'use strict';
    window.onblur = () => {};
    var video = document.getElementById("video_html5_api");
    video.muted = true;

    setTimeout(function() {
        video.play();
    }, 5000);

    video.onended = function() {
        console.log("Video bitti, bir sonrakine geçiliyor");
        document.getElementsByClassName("progress-icon")[0].parentElement.click();
        document.getElementById("video_html5_api").play();
    };

    setTimeout(function() {
        location.reload();
    }, 25*60*1000);
})();
